        function calculateSpeed() {
            const bandwidthMbps = parseFloat(document.getElementById('bandwidth').value);
            const wirelessStandard = document.getElementById('standard').value;
            const numDevices = parseInt(document.getElementById('devices').value);
            let expectedSpeed = 0;
            const resultElement = document.getElementById('result');

            if (isNaN(bandwidthMbps) || isNaN(numDevices) || numDevices <= 0) {
                resultElement.textContent = 'Please enter valid numbers for bandwidth and number of devices.';
                resultElement.style.display = 'block'; // Ensure the result is displayed
                return;
            }

            switch (wirelessStandard) {
                case '802.11ac':
                    expectedSpeed = (bandwidthMbps * 0.7) / numDevices;
                    break;
                case '802.11n':
                    expectedSpeed = (bandwidthMbps * 0.5) / numDevices;
                    break;
                default:
                    break;
            }

            resultElement.textContent = `Expected speed per device on ${wirelessStandard} network with ${numDevices} device(s): ${expectedSpeed.toFixed(2)} Mbps`;
            resultElement.style.display = 'block'; // Ensure the result is displayed
        }

        // Trigger calculation on input changes
        document.getElementById('bandwidth').addEventListener('input', calculateSpeed);
        document.getElementById('standard').addEventListener('change', calculateSpeed);
        document.getElementById('devices').addEventListener('input', calculateSpeed);

// Auto-detect new lines in the input
$taskInput.on("input", function () {
    var inputValue = $taskInput.val();

    // Check if there are newlines
    if (inputValue.includes("\n")) {
        var tasks = inputValue.split(/\n+/); // Split by newlines
        tasks.forEach(function (task) {
            if (task.trim() !== "") {
                addTask(task.trim()); // Add each note to the list
            }
        });
        $taskInput.val(""); // Clear the input field after processing
    }
});

// Add or update a task when pressing Enter
$taskInput.on("keypress", function (e) {
    if (e.which === 13 && $taskInput.val().trim() !== "") {
        e.preventDefault(); // Prevent new line in the input field
        var taskText = $taskInput.val().trim();

        // If editing, update the existing task
        if (editingTask) {
            editingTask.contents().get(0).nodeValue = taskText; // Update task text
            editingTask = null; // Clear the editing task
        } else {
            addTask(taskText); // Add new task if not editing
        }

        $taskInput.val(""); // Clear the input field
        displayNotification();
    }
});

<script>
    function parseDateInput(dateString) {
        const parts = dateString.match(/(\d{2})-(\w{3})-(\d{4}) (\d{2}):(\d{2}):(\d{2})/);
        if (!parts) return null;

        const [_, day, month, year, hours, minutes, seconds] = parts;
        const monthMap = {
            Jan: 0, Feb: 1, Mar: 2, Apr: 3,
            May: 4, Jun: 5, Jul: 6, Aug: 7,
            Sep: 8, Oct: 9, Nov: 10, Dec: 11
        };

        const monthIndex = monthMap[month];
        if (monthIndex === undefined) return null;

        return new Date(year, monthIndex, day, hours, minutes, seconds);
    }

    function calculateDateDifference() {
        const date1Str = document.getElementById("calc-date1").value.trim();
        const date2Str = document.getElementById("calc-date2").value.trim();
        
        const date1 = parseDateInput(date1Str);
        const date2 = parseDateInput(date2Str);

        const resultDiv = document.getElementById("calc-result");

        if (!date1 || !date2) {
            resultDiv.innerText = "Invalid date format. Please use DD-MMM-YYYY HH:MM:SS";
            resultDiv.classList.add("calc-error-unique");
            return;
        }

        if (date1 > date2) {
            resultDiv.innerText = "Start date must be before end date.";
            resultDiv.classList.add("calc-error-unique");
            return;
        }

        let months = date2.getMonth() - date1.getMonth() + (12 * (date2.getFullYear() - date1.getFullYear()));
        let days = date2.getDate() - date1.getDate();

        if (days < 0) {
            months--;
            const lastMonthDate = new Date(date1.getFullYear(), date1.getMonth() + 1, 0);
            days += lastMonthDate.getDate();
        }

        resultDiv.innerText = `Difference: ${months} months and ${days} days`;
        resultDiv.classList.remove("calc-error-unique");
    }

    // Automatically trigger calculation on input
    document.getElementById("calc-date1").addEventListener("input", calculateDateDifference);
    document.getElementById("calc-date2").addEventListener("input", calculateDateDifference);
</script>
